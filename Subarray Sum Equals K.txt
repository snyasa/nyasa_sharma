Subarray Sum Equals K

class Solution {
    public int subarraySum(int[] nums, int k) {
        int sum=0;
        int count=0;
        HashMap<Integer,Integer>map=new HashMap<>();
        for(int i=0;i<nums.length;i++){
            sum=sum+nums[i];
            if(sum==k){
                count++;
            }
            if(map.get(sum-k)){
                count++;
            }
            map.put(sum,1);
        }
        return count;
    }
}
